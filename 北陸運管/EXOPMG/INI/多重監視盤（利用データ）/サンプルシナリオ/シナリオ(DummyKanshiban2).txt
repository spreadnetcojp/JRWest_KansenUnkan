	000-000-0000-00 ActiveOne 接続初期化要求.dat,, 60s, OnError, OnError, OnError, False
	#OnComStarted
	000-000-0000-00 RegTimerProc 0, 0, 30s, 利用データ送信処理

	000-000-0000-00 WaitUntil 05:00:00
